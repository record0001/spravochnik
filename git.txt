

git config --global user.name "Your Name" # указать имя, которым будут подписаны коммиты
git config --global user.email "e@w.com"  # указать электропочту, которая будет в описании коммитера


git clone http://		Ссылка откуда клонировать


git status			Статус текущего локального репозитория


git add имя файла		Добавляет файл для отслеживания антрэкнуть


git add.			Добавляет все новые файлы для отслеживания антрэкнуть


git commit -am "Комментарий"	Зафиксировть изменения в файле и закомментировать закомитить


git push			Отправить все закоммиченные изменения в репозиторий на сервер запушить


git pull			Забрать все измененные файлы с сервера



Перед каждым пушем коммиттеть

При добавлении файлов для каждого писать git add имя файла

Все новые файлы коммитить
